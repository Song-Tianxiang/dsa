// 插入排序
func sortArray(nums []int) []int {
    for j := 1; j < len(nums); j++ {
        for i := j-1; i >= 0; i-- {
            if nums[i] > nums[i+1] {
                nums[i], nums[i+1] = nums[i+1], nums[i]
            } else {
                break
            }
        }
    }

    return nums
}
